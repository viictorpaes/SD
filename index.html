Plague

Desafie a sorte: um jogo de tensão crescente onde cartas definem seu destino e um botão errado aciona o alarme.

Ver no GitHub

Um jogo estilo 'Croc Croc'. Serão uma sequência de botões e os jogadores puxarão cartas cada um para definir quantos botões apertar. Na rodada 1 será um botão infectado, na 2 dois botões e assim sucessivamente. Quando o botão correto for pressionado um led verde acenderá e quando o infectado for pressionado um led vermelho seguido de um beep para indicar que o jogador perdeu.

Código

//Declarção das entradas:
int vermPin1 = 9;     // Pino do LED vermelho do primeiro conjunto
int verdPin1 = 8;     // Pino do LED verde do primeiro conjunto
int btnPin1  = 2;     // Pino do botão do primeiro conjunto

int vermPin2 = 11;    // Pino do LED vermelho do segundo conjunto
int verdPin2 = 10;    // Pino do LED verde do segundo conjunto
int btnPin2  = 3;     // Pino do botão do segundo conjunto

int vermPin3 = 13;    // Pino do LED vermelho do terceiro conjunto
int verdPin3 = 12;    // Pino do LED verde do terceiro conjunto
int btnPin3  = 4;     // Pino do botão do terceiro conjunto

int btnReset = 5;     // Pino do botão de reset

int buzzer = 6;       // Pino do buzzer

int sorteio;          // Variável que armazena o número aleatório (1 a 3)

void setup() {  //essa função é rodada pelo arduino uma unica vez antes de iniciar o código
  Serial.begin(9600);           // Inicializa comunicação serial

  pinMode(vermPin1, OUTPUT);    // Define o LED vermelho 1 como saída
  pinMode(verdPin1, OUTPUT);    // Define o LED verde 1 como saída
  pinMode(vermPin2, OUTPUT);    // Define o LED vermelho 2 como saída
  pinMode(verdPin2, OUTPUT);    // Define o LED verde 2 como saída
  pinMode(vermPin3, OUTPUT);    // Define o LED vermelho 3 como saída
  pinMode(verdPin3, OUTPUT);    // Define o LED verde 3 como saída

  pinMode(btnPin1, INPUT);      // Define o botão 1 como entrada
  pinMode(btnPin2, INPUT);      // Define o botão 2 como entrada
  pinMode(btnPin3, INPUT);      // Define o botão 3 como entrada
  
  pinMode(buzzer, OUTPUT);      // Define o buzzer como saída

  pinMode(btnReset, INPUT);     // Define o botão de reset como entrada

  randomSeed(analogRead(A0));   // Gera semente aleatória 
  sorteio = random(1, 4);       // Sorteia um número de 1 a 3
}

void desligaTodos() { //função que eu criei para usar depois
  digitalWrite(vermPin1, LOW);  // Desliga LED vermelho 1
  digitalWrite(verdPin1, LOW);  // Desliga LED verde 1
  digitalWrite(vermPin2, LOW);  // Desliga LED vermelho 2
  digitalWrite(verdPin2, LOW);  // Desliga LED verde 2
  digitalWrite(vermPin3, LOW);  // Desliga LED vermelho 3
  digitalWrite(verdPin3, LOW);  // Desliga LED verde 3

  randomSeed(analogRead(A0));   // Atualiza a semente aleatória
  sorteio = random(1, 4);       // Sorteia novamente um número entre 1 e 3
}

void loop() { //essa função o arduino fica rodando de novo e de novo

  int b1 = digitalRead(btnPin1);   // Lê estado do botão 1
  int b2 = digitalRead(btnPin2);   // Lê estado do botão 2
  int b3 = digitalRead(btnPin3);   // Lê estado do botão 3
  int bReset = digitalRead(btnReset); // Lê estado do botão de reset

  if (bReset == HIGH) {           // Se o botão de reset for pressionado
    desligaTodos();               // Desliga todos LEDs e faz novo sorteio
  }

  // Caso o número sorteado seja 1
  if (sorteio == 1) {

    if (b1 == HIGH){              // Se o botão 1 foi pressionado
        digitalWrite(vermPin1, HIGH); // Liga LED vermelho 1
        tone(buzzer, 1500);           // Aciona o buzzer
        delay(300);                   // Aguarda 300 ms
        noTone(buzzer);               // Desliga o buzzer
    }

    if (b2 == HIGH) digitalWrite(verdPin2, HIGH); // Pressão no botão 2 → LED verde 2 liga
    if (b3 == HIGH) digitalWrite(verdPin3, HIGH); // Pressão no botão 3 → LED verde 3 liga
  }

  // Caso o número sorteado seja 2
  else if (sorteio == 2) {

    if (b1 == HIGH) digitalWrite(verdPin1, HIGH); // Pressão no botão 1 → LED verde 1 liga

    if (b2 == HIGH){                // Pressão no botão 2
        digitalWrite(vermPin2, HIGH); // Liga LED vermelho 2
        tone(buzzer, 1500);           // Toca o buzzer
        delay(300);
        noTone(buzzer);
    }

    if (b3 == HIGH) digitalWrite(verdPin3, HIGH); // Pressão no botão 3 → LED verde 3 liga
  }

  // Caso o número sorteado seja 3
  else if (sorteio == 3) {

    if (b1 == HIGH) digitalWrite(verdPin1, HIGH); // Pressão no botão 1 → LED verde 1 liga
    if (b2 == HIGH) digitalWrite(verdPin2, HIGH); // Pressão no botão 2 → LED verde 2 liga

    if (b3 == HIGH){                // Pressão no botão 3
        digitalWrite(vermPin3, HIGH); // Liga LED vermelho 3
        tone(buzzer, 1500);
        delay(300);
        noTone(buzzer);
    }
  }
}
